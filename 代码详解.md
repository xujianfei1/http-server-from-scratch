## 四、认证函数（第67-92行）

### check_auth() 函数

```python
def check_auth(headers):
    """检查 HTTP Basic Auth 认证"""
    auth_header = headers.get('Authorization', '')
    # 从请求头获取 Authorization 字段

    if not auth_header.startswith('Basic '):
        return False
    # 没有 Basic 认证信息，返回失败

    encoded = auth_header[6:]
    # 去掉 "Basic " 前缀，得到编码后的字符串
    # 例如："Basic YWRtaW46MTIzNDU2"

    try:
        decoded = base64.b64decode(encoded).decode('utf-8')
        # Base64 解码："YWRtaW46MTIzNDU2" → "admin:123456"
        username, password = decoded.split(':', 1)
        # 分割用户名和密码
        return username == CONFIG['username'] and password == CONFIG['password']
        # 对比配置中的用户名密码
    except:
        return False
```

**调用**：`handle_request()` 第165行调用，所有请求都需要认证

---

### make_auth_response() 函数

```python
def make_auth_response():
    """返回 401 认证挑战响应"""
    body = "<h1>401 Unauthorized</h1><p>需要登录</p>"
    body_bytes = body.encode('utf-8')

    response_line = "HTTP/1.1 401 Unauthorized\r\n"
    # 状态码 401 表示需要认证

    response_headers = "WWW-Authenticate: Basic realm=\"HTTP Server\"\r\n"
    # 告诉浏览器使用 Basic Auth 认证
    # realm 是认证域的名称，浏览器会显示

    response_headers += "Content-Type: text/html; charset=utf-8\r\n"
    response_headers += f"Content-Length: {len(body_bytes)}\r\n"
    response_headers += "Connection: close\r\n"
    response_headers += "\r\n"

    response = response_line.encode() + response_headers.encode() + body_bytes
    return response
```

**调用**：`handle_request()` 第168行，认证失败时返回 401

---

### HTTP Basic Auth 流程

```
浏览器请求（无认证）
    ↓
check_auth() 返回 False
    ↓
make_auth_response() 返回 401 + WWW-Authenticate 头
    ↓
浏览器弹窗要求输入用户名密码
    ↓
浏览器重新发送请求（带 Authorization 头）
    ↓
check_auth() 验证通过，继续处理请求
```

## 七、工具函数（第94-147行）

### parse_request() 解析请求

```python
def parse_request(request_text):
    lines = request_text.split('\r\n')
    # 把请求文本按换行分割
    # ["GET / HTTP/1.1", "Host: localhost", ...]

    request_line = lines[0]      # 第一行：请求行
    parts = request_line.split(' ')
    # ["GET", "/", "HTTP/1.1"]

    method = parts[0]            # "GET"
    path = parts[1]              # "/"
    version = parts[2]           # "HTTP/1.1"

    headers = {}                  # 存放请求头
    for line in lines[1:]:       # 遍历剩下的行
        if line == '':           # 遇到空行就结束
            break
        key, value = line.split(': ', 1)
        # "Host: localhost" → key="Host", value="localhost"
        headers[key] = value

    return method, path, version, headers
```

**调用**：`handle_request()` 第163行调用

### make_response() 构建响应

```python
def make_response(body, status=200, content_type="text/html"):
    status_text = "OK" if status == 200 else "Not Found"
    # 状态码 200 → "OK"，否则 → "Not Found"

    body_bytes = body.encode('utf-8')
    # 字符串转字节，因为网络传输用字节

    response_line = f"HTTP/1.1 {status} {status_text}\r\n"
    # "HTTP/1.1 200 OK\r\n"

    response_headers = f"Content-Type: {content_type}; charset=utf-8\r\n"
    response_headers += f"Content-Length: {len(body_bytes)}\r\n"
    response_headers += "Connection: close\r\n"
    response_headers += "\r\n"
    # 每个 header 后要 \r\n，最后空行分隔 header 和 body

    response = response_line.encode() + response_headers.encode() + body_bytes
    # 拼接成完整响应

    return response
```

## 八、请求处理（第149-205行）

### handle_request() 核心函数

```python
def handle_request(client_socket, address):
    start_time = datetime.now()           # 记录开始时间

    try:
        data = client_socket.recv(8192)   # 接收请求（最多8KB）
        # recv 是阻塞的，等客户端发数据

        if not data:                      # 如果没收到数据
            return                        # 直接返回

        request_text = data.decode('utf-8', errors='replace')
        method, path, version, headers = parse_request(request_text)
        # 解析请求

        log(f"{address} {method} {path}") # 记录日志

        # 安全检查：认证验证
        if not check_auth(headers):
            log(f"{address} 认证失败")
            response = make_auth_response()
            client_socket.send(response)
            return

        # ===== 判断是静态文件还是动态路由 =====
        if path in ['/index.html', '/style.css', '/script.js'] or path.startswith('/static/'):
            # 安全检查
            if '..' in path:              # 防止路径遍历攻击
                response = make_response("<h1>403...</h1>", 403)
                client_socket.send(response)
                return

            file_path = path.lstrip('/')  # 去掉开头的 /
            full_path = PUBLIC_DIR / file_path
            # 拼接完整路径

            try:
                response = make_file_response(full_path)
                client_socket.send(response)
            except FileNotFoundError:
                response = make_response("<h1>404...</h1>", 404)
                client_socket.send(response)
            return

        # 动态路由
        handler = ROUTES.get(path, not_found_handler)
        # 从路由表找处理器，找不到就用 not_found_handler
        body = handler()                  # 调用处理器函数
        response = make_response(body)    # 构建响应
        client_socket.send(response)       # 发送响应

    except Exception as e:                 # 任何错误
        log(f"{address} 错误: {e}")       # 记录错误
        try:
            response = make_response("<h1>500...</h1>", 500)
            client_socket.send(response)   # 尝试返回错误页面
        except:
            pass                          # 忽略发送失败

    finally:                              # 不管成功还是失败
        client_socket.close()             # 关闭连接
        duration = (datetime.now() - start_time).total_seconds() * 1000
        log(f"{address} 关闭 (耗时 {duration:.1f}ms)")
```

**调用**：`main()` 第233行，每来一个连接启动一个新线程调用

## 九、主循环（第207-242行）

### main() 服务器主函数

```python
def main():
    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    # AF_INET = IPv4，SOCK_STREAM = TCP

    server.bind((CONFIG["host"], CONFIG["port"]))
    # 绑定地址和端口

    server.listen(5)
    # 开始监听，最多排队5个连接

    # ===== 打印启动界面 =====
    print("=" * 50)
    print("  HTTP Server from Scratch")
    ...
    print(f"  端口: {CONFIG['port']}")
    print("按 Ctrl+C 停止服务器")

    log("服务器启动")

    # ===== 无限循环接收连接 =====
    while True:
        try:
            client, address = server.accept()
            # accept 阻塞，等待客户端连接
            # 返回 (客户端socket, 客户端地址)

            log(f"{address} 新连接")

            # 启动新线程处理请求
            thread = threading.Thread(target=handle_request, args=(client, address))
            thread.start()
            # 主线程继续循环，接收下一个连接

        except KeyboardInterrupt:          # Ctrl+C
            log("服务器关闭")
            print("\n服务器已停止")
            break                          # 退出循环

        except Exception as e:            # 其他错误
            log(f"服务器错误: {e}")
```

## 关键概念解释

### Socket 是什么？

Socket 是**网络通信的端点**，就像电话机。服务器创建 Socket 并监听端口，客户端连接到这个端口进行通信。

### HTTP 请求格式

```
GET /hello HTTP/1.1          ← 请求行（方法 路径 协议）
Host: localhost:8080         ← 请求头
User-Agent: curl/7.68.0
Accept: */*
                               ← 空行
（可选）请求体
```

### HTTP 响应格式

```
HTTP/1.1 200 OK              ← 状态行（协议 状态码 状态文本）
Content-Type: text/html      ← 响应头
Content-Length: 123
Connection: close
                               ← 空行
<html>...</html>              ← 响应体
```

### 什么是 MIME 类型？

MIME 类型告诉浏览器文件的格式，让浏览器知道怎么解析和显示。比如：
- `text/html` → 网页
- `image/png` → 图片
- `application/json` → JSON 数据

### 多线程的作用？

单线程时，一个请求处理完才能接收下一个。加入多线程后，每个请求在独立线程中处理，可以同时响应多个用户。

### 什么是 HTTP Basic Auth？

HTTP Basic Auth 是一种简单的身份验证方式：

```
1. 客户端请求受保护的资源
2. 服务器返回 401 + WWW-Authenticate 头
3. 浏览器弹窗要求输入用户名密码
4. 浏览器重新请求，携带 Authorization 头
5. 服务器验证通过，返回资源
```

**Authorization 头格式**：

```text
Authorization: Basic base64("用户名:密码")
# 例如：admin:123456 → YWRtaW46MTIzNDU2
```

**优点**：简单、实现方便
**缺点**：密码 Base64 编码（可逆），建议配合 HTTPS 使用
